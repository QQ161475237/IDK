-- ==================== Gamepass 解锁脚本（监听传送版） ====================

-- 你的 GitHub 链接
local SCRIPT_URL = "https://raw.githubusercontent.com/QQ161475237/IDK/main/%E5%A2%A8%E6%B0%B4%E6%B8%B8%E6%88%8FFE%E9%80%9A%E8%A1%8C%E8%AF%81.txt"

-- 标记是否已注册监听
if not getgenv()._gamepass_listener_registered then
    getgenv()._gamepass_listener_registered = true
    
    -- 监听服务器切换
    local Players = game:GetService("Players")
    local LocalPlayer = Players.LocalPlayer
    
    local function onTeleport()
        -- 延迟执行，确保新服务器加载完成
        task.wait(1)
        -- 重新加载脚本
        loadstring(game:HttpGet(SCRIPT_URL))()
    end
    
    -- 方法1：监听 OnTeleport 事件
    pcall(function()
        LocalPlayer.OnTeleport:Connect(onTeleport)
        print("[Gamepass] 已注册 OnTeleport 监听")
    end)
    
    -- 方法2：使用 queue_on_teleport（备用）
    local queue_func = syn and syn.queue_on_teleport or queue_on_teleport or (fluxus and fluxus.queue_on_teleport)
    if queue_func then
        pcall(function()
            queue_func("loadstring(game:HttpGet('" .. SCRIPT_URL .. "'))()")
            print("[Gamepass] 已注册 queue_on_teleport")
        end)
    end
end

-- ==================== 执行解锁 ====================
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- 检查是否已解锁（避免重复）
if LocalPlayer:GetAttribute("__OwnsPermGuard") == true then
    print("[Gamepass] 当前服务器已解锁，跳过")
    return
end

local GAMEPASS_CONFIG = {
    ["PERMANENT_GUARD"]       = { Bought = true, GamepassId = 0 },
    ["EMOTE_PAGES"]           = { Bought = true, GamepassId = 0 },
    ["VIP"]                   = { Bought = true, GamepassId = 0 },
    ["PRIVATE_SERVER_PLUS"]   = { Bought = true, GamepassId = 0 },
    ["GLASS_VISION"]          = { Bought = true, GamepassId = 0 },
    ["VOTE_2X"]               = { Bought = true, GamepassId = 0 },
    ["CUSTOM_TAG"]            = { Bought = true, GamepassId = 0 }
}

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local StarterGui = game:GetService("StarterGui")

local function GrantGamepass(configName, handlerFunc)
    local config = GAMEPASS_CONFIG[configName]
    if config and config.Bought then
        task.spawn(handlerFunc, config)
    end
end

GrantGamepass("PERMANENT_GUARD", function(config)
    local args = { [1] = { Time = 0, PlayerName = LocalPlayer.Name, Action = "Purchase", Permanent = true, Role = "Guard" } }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__OwnsPermGuard", true)
end)

GrantGamepass("EMOTE_PAGES", function(config)
    local args = { [1] = { EmoteSlots = 5, ExtraPages = 2, Action = "RefreshEmotes", PlayerName = LocalPlayer.Name } }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__OwnsEmotePages", true)
end)

GrantGamepass("VIP", function(config)
    local args = {
        [1] = {
            VIP = true, Multipliers = { Luck = 2, Won = 2, XP = 2 },
            Tag = "VIP", Action = "VIP", CustomOutfit = true, PlayerName = LocalPlayer.Name
        }
    }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("C_OwnsVIPGamepass", true)
end)

GrantGamepass("PRIVATE_SERVER_PLUS", function(config)
    local args = { [1] = { PrivateServer = true, Commands = true, Action = "PrivateServer", PlayerName = LocalPlayer.Name } }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__OwnsPSPlus", true)
end)

GrantGamepass("GLASS_VISION", function(config)
    local args = {
        [1] = {
            GlassBridge = { Vision = "Manufacturer" }, EffectName = "RevealGlassTiles",
            Permanent = true, Interval = 2, Action = "GlassBridge", PlayerName = LocalPlayer.Name
        }
    }
    ReplicatedStorage.Remotes.Effects:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__OwnsGlassManufacturerVision", true)
end)

GrantGamepass("VOTE_2X", function(config)
    local args = { [1] = { VoteMultiplier = 2, VotingPower = 2, Action = "VoteMultiplier", PlayerName = LocalPlayer.Name } }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__Owns2XVoteGamepass", true)
end)

GrantGamepass("CUSTOM_TAG", function(config)
    local args = { [1] = { CustomTag = true, TagSelection = true, Action = "CustomTag", PlayerName = LocalPlayer.Name } }
    ReplicatedStorage.Remotes.PlayableGuardRemote:FireServer(unpack(args))
    LocalPlayer:SetAttribute("__OwnsCustomPlayerTag", true)
end)

local TargetAttributes = {
    "__OwnsPermGuard", "__OwnsEmotePages", "__OwnsVIPGamepass",
    "__OwnsPSPlus", "__OwnsGlassManufacturerVision", "__Owns2XVoteGamepass", "__OwnsCustomPlayerTag"
}

for _, attrName in ipairs(TargetAttributes) do
    LocalPlayer:SetAttribute(attrName, true)
end

StarterGui:SetCore("SendNotification", {
    Title = "SUCCESS",
    Text = "Gamepass 已解锁\n传送后自动重新执行",
    Duration = 5
})

print("[Gamepass] 解锁完成，监听器已注册")
